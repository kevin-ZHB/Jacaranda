class GC: 
    level = 1
    exp = 0
    
    def __init__(self, name, age):
        self.name = name
        self.age = age

    def attack(self):
        print(f"{self.name} attacked!")
        
    def watch(self):
        print(f"{self.name} stares at you O.o")
        
class hop(GC):
    def jump(self):
        print(f"{self.name} jumpped!!!")

player1 = hop("john doe", 99)
player1.attack()
player1.jump()

john doe attacked!
